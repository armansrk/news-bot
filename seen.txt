def load_seen() -> set:
    try:
        if not os.path.exists("seen.txt"):
            return set()
        with open("seen.txt", "r", encoding="utf-8") as f:
            seen = set(line.strip() for line in f if line.strip())
        print(f"مطالب دیده شده: {seen}")
        return seen
    except Exception as e:
        print(f"خطا در بارگذاری فایل: {e}")
        return set()
